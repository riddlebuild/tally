// ============================================================ // Tally — staff-login Edge Function (v3: adds staff name+PIN mode) // // Handles THREE login types, all server-side with lockout: // 1. Staff by name: { mode:"staff_name", name, pin } <-- NEW (typed login) // 2. Viewer: { mode:"viewer", pin } (secret door) // 3. Staff by id: { staff_id, pin } (legacy fallback) // // Deploy into the SAME `staff-login` function (replace previous code). // Keep "Verify JWT" OFF. Uses the auto-injected SUPABASE_SERVICE_ROLE_KEY // (bypasses RLS), so it keeps working after the staff table is locked. // ============================================================ import { createClient } from "jsr:@supabase/supabase-js@2"; const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", }; const MAX_FAILS = 5; const LOCK_MINUTES = 15; function json(body: unknown, status = 200) { return new Response(JSON.stringify(body), { status, headers: { ...corsHeaders, "Content-Type": "application/json" }, }); } Deno.serve(async (req) => { if (req.method === "OPTIONS") return new Response("ok", { headers: corsHeaders }); try { const { staff_id, pin, mode, name } = await req.json().catch(() => ({})); if (!pin) return json({ error: "Missing PIN." }, 400); const url = Deno.env.get("SUPABASE_URL"); const serviceKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") || Deno.env.get("SUPABASE_SECRET_KEY"); if (!url || !serviceKey) return json({ error: "Server is missing its service key." }, 500); const admin = createClient(url, serviceKey, { auth: { persistSession: false } }); // 1. Resolve the staff row let staff: any = null; if (mode === "viewer") { const { data: viewers } = await admin .from("staff") .select("id,name,pin,role,outlet_id,tenant_id,auth_user_id") .eq("role", "viewer").limit(1); staff = viewers && viewers[0]; if (!staff) return json({ error: "Viewer access is not set up yet." }, 404); } else if (mode === "staff_name") { const nm = (name || "").trim(); if (!nm) return json({ error: "Missing name." }, 400); // Case-insensitive exact name match; viewers can't log in this way. const { data: matches } = await admin .from("staff") .select("id,name,pin,role,outlet_id,tenant_id,auth_user_id") .neq("role", "viewer") .ilike("name", nm); if (!matches || matches.length === 0) return json({ error: "Name not found. Check spelling." }, 404); // If duplicate names exist, prefer the one whose PIN matches. staff = matches.find((m: any) => String(m.pin) === String(pin)) || matches[0]; } else { // Legacy: lookup by staff_id (kept for backward compatibility) if (!staff_id) return json({ error: "Missing name or PIN." }, 400); const { data: s } = await admin .from("staff") .select("id,name,pin,role,outlet_id,tenant_id,auth_user_id") .eq("id", staff_id).maybeSingle(); staff = s; if (!staff) return json({ error: "Staff not found." }, 404); } const sid = staff.id; // 2. Lockout check (keyed by the resolved staff id) const { data: attempt } = await admin .from("pin_attempts").select("*").eq("staff_id", sid).maybeSingle(); if (attempt?.locked_until && new Date(attempt.locked_until) > new Date()) { return json({ error: "Too many wrong tries. Please wait a few minutes." }, 429); } // 3. Verify PIN if (String(staff.pin) !== String(pin)) { const fails = (attempt?.fail_count || 0) + 1; const locked = fails >= MAX_FAILS; await admin.from("pin_attempts").upsert({ staff_id: sid, fail_count: locked ? 0 : fails, locked_until: locked ? new Date(Date.now() + LOCK_MINUTES * 60000).toISOString() : null, }, { onConflict: "staff_id" }); return json({ error: locked ? `Locked for ${LOCK_MINUTES} minutes after too many tries.` : "Incorrect PIN.", }, 401); } // PIN correct -> clear failed-attempt record await admin.from("pin_attempts").upsert( { staff_id: sid, fail_count: 0, locked_until: null }, { onConflict: "staff_id" }); // 4. Ensure a real auth identity for this staff/viewer const email = `staff_${sid}@staff.harulabs.io`; let authUserId = staff.auth_user_id; if (!authUserId) { const randomPwd = crypto.randomUUID() + crypto.randomUUID(); const { data: created, error: cErr } = await admin.auth.admin.createUser({ email, password: randomPwd, email_confirm: true, user_metadata: { staff_id: sid, name: staff.name, role: staff.role }, }); if (cErr || !created?.user) { return json({ error: "Could not set up account: " + (cErr?.message || "unknown") }, 500); } authUserId = created.user.id; await admin.from("staff").update({ auth_user_id: authUserId }).eq("id", sid); } // Keep the tenant link current (drives tenant isolation) await admin.from("tenant_users").upsert( { user_id: authUserId, tenant_id: staff.tenant_id, role: staff.role }, { onConflict: "user_id" }); // 5. Mint a real session (admin generateLink -> verifyOtp) const { data: linkData, error: lErr } = await admin.auth.admin.generateLink({ type: "magiclink", email, }); if (lErr || !linkData?.properties?.hashed_token) { return json({ error: "Could not start session (link)." }, 500); } const { data: otp, error: oErr } = await admin.auth.verifyOtp({ token_hash: linkData.properties.hashed_token, type: "magiclink", }); if (oErr || !otp?.session) { return json({ error: "Could not start session (otp)." }, 500); } return json({ ok: true, access_token: otp.session.access_token, refresh_token: otp.session.refresh_token, user: { staff_id: staff.id, name: staff.name, role: staff.role, outlet_id: staff.outlet_id, tenant_id: staff.tenant_id, }, }); } catch (e) { return json({ error: "Server error: " + ((e as Error)?.message || String(e)) }, 500); } });
